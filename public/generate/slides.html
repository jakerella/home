function denodify( nodeStyleFn ) {
    return function promiseStyleFn(originalArg) {

        return new Promise(function( resolve, reject ) {

            nodeStyleFn(originalArg, function(err, value) {

                if (err) { return reject(err); }
                resolve(value);

            });
        });
    };
